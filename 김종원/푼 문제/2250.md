import java.util.*;
import java.io.*;
import java.util.stream.*;

public class Main {
    
    static int[][] nodes;
    static ArrayList<ArrayList<Integer>> dist;
    static int maxDepth;
    static int index;
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        
        int N = Integer.parseInt(br.readLine());
        
        boolean[] rootFind = new boolean[N+1];
        
        nodes = new int[N+1][2];
        dist = new ArrayList<>();
        for(int i = 0; i < N; i++) {
            StringTokenizer st = new StringTokenizer(br.readLine(), " ");
            int root = Integer.parseInt(st.nextToken());
            int left = Integer.parseInt(st.nextToken());
            int right = Integer.parseInt(st.nextToken());
            
            nodes[root][0] = left;
            nodes[root][1] = right;
            if(left != -1)
                rootFind[left] = true;
            if(right != -1)
                rootFind[right] = true;
        }
        
        int rootNo = 0;
        for(int i = 1; i < N+1; i++) {
            if(!rootFind[i]) {
                rootNo = i;
                break;
            }
        }
        
        maxDepth = 1;
        index = 0;
        dist.add(new ArrayList<>());
        dist.add(new ArrayList<>());
        dfs(rootNo, maxDepth);
        
        int maxLen = 0;
        int maxIdx = 0;
        for(int i = 1; i <= maxDepth; i++) {
            List<Integer> list = dist.get(i).stream().sorted().collect(Collectors.toList());
            
            int size = list.get(list.size() - 1) - list.get(0) + 1;
            if(size > maxLen) {
                maxIdx = i;
                maxLen = size;
            }
        }
        
        System.out.println(maxIdx + " " + maxLen);
    }
    
    public static void dfs(int root, int depth) {
        if(depth > maxDepth) {
            maxDepth = depth;
            dist.add(new ArrayList<>());
        }
        
        
        if(nodes[root][0] != -1) {
            dfs(nodes[root][0], depth + 1);
        }
        dist.get(depth).add(index++);
        if(nodes[root][1] != -1) {
            dfs(nodes[root][1], depth + 1);
        }
    }
}
