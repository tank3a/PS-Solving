//4시 30분
import java.io.*;
import java.util.*;

public class Main {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        
        int flag = 1;
        
        String input = br.readLine();
        
        int prev = 0;
        int answer = 0;
        for(int i = 0; i < input.length(); i++) {
            if(input.charAt(i) == '-') {
                int next = Integer.parseInt(input.substring(prev, i));
                answer += flag * next;
                prev = i + 1;
                flag = -1;
                continue;
            }
            if(input.charAt(i) == '+') {
                int next = Integer.parseInt(input.substring(prev, i));
                answer += flag * next;
                prev = i + 1;
                continue;
            }
            
        }
        
        answer += flag * Integer.parseInt(input.substring(prev));
        
        System.out.println(answer);
    }
}
