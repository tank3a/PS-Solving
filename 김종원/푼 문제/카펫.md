class Solution {
    public int[] solution(int brown, int yellow) {
        int[] answer = new int[2];
        
        for(int i = 1; i <= Math.sqrt(yellow); i++) {
            if(yellow % i != 0)
                continue;
            
            int row = i;
            int col = yellow / i;
            
            if(row * 2 + col * 2 + 4 == brown)  {
                answer = new int[]{col + 2, row + 2};
                break;
            }
        }
        return answer;
    }
}
