import java.util.Stack;
import java.util.Arrays;
import java.util.StringTokenizer;

class Solution {
    public String[] solution(String[][] plans) {
        int answerIdx = 0;
        String[] answer = new String[plans.length];
        Stack<Homework> stack = new Stack<>();
        Homework[] list = new Homework[plans.length];
        
        for(int i = 0; i < plans.length; i++) {
            StringTokenizer st = new StringTokenizer(plans[i][1], ":");
            Homework hw = new Homework(
                plans[i][0], 
                Integer.parseInt(st.nextToken()), 
                Integer.parseInt(st.nextToken()), 
                Integer.parseInt(plans[i][2]));
            
            list[i] = hw;
        }
        
        Arrays.sort(list);
        
        int curH = list[0].h;
        int curM = list[0].m;
        stack.push(list[0]);
        
        
        for(int i = 1; i < list.length; i++) {
            Homework now = list[i];
            int diffH = now.h - curH;
            int diffM = now.m - curM;
            if(diffM < 0) {
                diffH--;
                diffM += 60;
            }
            
            diffM += diffH * 60;
            while(diffM > 0 && !stack.isEmpty()) {
                Homework prev = stack.pop();

                if(prev.time >= diffM) {
                    prev.time -= diffM;
                    if(prev.time != 0)
                        stack.push(prev);
                    else {
                        answer[answerIdx++] = prev.subject;
                    }
                    diffM = 0;
                    break;
                }
                
                diffM -= prev.time;
                answer[answerIdx++] = prev.subject;
            }
            
            curH = now.h;
            curM = now.m;
            stack.push(now);
            
        }
        
        while(!stack.isEmpty()) {
            answer[answerIdx++] = stack.pop().subject;
        }
        
        return answer;
    }
}

class Homework implements Comparable<Homework> {
    String subject;
    int h;
    int m;
    int time;
    
    public Homework(String subject, int h, int m, int time) {
        this.subject = subject;
        this.h = h;
        this.m = m;
        this.time = time;
    }
    
    @Override
    public int compareTo(Homework o) {
        if(o.h == this.h) {
            return this.m - o.m;
        }
        return this.h - o.h;
    }
}
