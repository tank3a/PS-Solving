import java.util.*;
import java.io.*;

public class Main {
    public static void main(String[] argrs) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        
        String input = br.readLine();
        
        String[] answer = new String[8];
        
        int index = -1;
        int count = 0;
        int zeros = 4;
        StringBuilder now = new StringBuilder();
        for(int i = 0; i < input.length(); i++) {
            
            if(input.charAt(i) != ':') {
                now.append(input.charAt(i));
                zeros--;
                continue;
            }
            
            if(i+1 < input.length()) {
                if(input.charAt(i+1) == ':') {
                    if(now.length() != 0) {
                        StringBuilder newNow = new StringBuilder();
                        for(int j = 0; j < zeros; j++)
                            newNow.append('0');
                        newNow.append(now);

                        answer[count] = newNow.toString();
                        count++;
                        zeros = 4;
                        now = new StringBuilder();
                    }
                    index = count++;
                    i++;
                    continue;
                }
            }

            StringBuilder newNow = new StringBuilder();
            for(int j = 0; j < zeros; j++)
                newNow.append('0');
            newNow.append(now);
            
            answer[count] = newNow.toString();
            count++;
            zeros = 4;
            now = new StringBuilder();
        }
        
        StringBuilder sb = new StringBuilder();

        if(now.length() != 0) {
            StringBuilder newNow = new StringBuilder();
            for(int j = 0; j < zeros; j++)
                newNow.append('0');
            newNow.append(now);

            answer[count] = newNow.toString();
            count++;
        }
        for(int i = 0; i < count; i++) {
            if(i == index) {
                for(int j = 0; j <= 8 - count; j++)
                    sb.append("0000:");
                continue;
            }
            sb.append(answer[i]).append(':');

        }
        
        if(count == 0) {
            for(int i = 0; i < 8; i++)
                sb.append("0000:");
        }
        
        sb.deleteCharAt(sb.length() - 1);
        
        System.out.println(sb.toString());
    }
}
