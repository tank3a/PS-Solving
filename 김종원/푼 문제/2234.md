import java.util.*;
import java.io.*;

public class Main {
    
    static int M, N;
    static int[][] matrix;
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        
        StringTokenizer st = new StringTokenizer(br.readLine(), " ");
        
        N = Integer.parseInt(st.nextToken());
        M = Integer.parseInt(st.nextToken());
        
        matrix = new int[M][N];
        
        for(int i = 0; i < M; i++) {
            StringTokenizer input = new StringTokenizer(br.readLine(), " ");
            
            for(int j = 0; j < N; j++) {
                matrix[i][j] = Integer.parseInt(input.nextToken());
            }
        }
        
        int[][] check = new int[M][N];
        HashMap<Integer, Integer> counts = new HashMap<>();
        int index = 1;
        for(int i = 0; i < M; i++) {
            for(int j = 0; j < N; j++) {
                if(check[i][j] != 0)
                    continue;
                
                Queue<int[]> queue = new LinkedList<>();
                
                queue.add(new int[]{i, j});
                int count = bfs(queue, check, index);
                counts.put(index, count);
                index++;
            }
        }
        
        StringBuilder sb = new StringBuilder();
        sb.append(index - 1).append('\n');
        
        int max = 0;
        for(int i = 1; i < index; i++) {
            if(max < counts.get(i))
                max = counts.get(i);
        }
        
        sb.append(max).append('\n');
        
        ArrayList<HashSet<Integer>> list = new ArrayList<>();
        for(int i = 0; i < index; i++)
            list.add(new HashSet<>());
        for(int i = 0; i < M; i++) {
            for(int j = 0; j < N; j++) {
                if(i + 1 < M) {
                    int now = check[i][j];
                    int next = check[i+1][j];
                    
                    if(now != next) {
                        list.get(now).add(next);
                        list.get(next).add(now);
                    }
                }
                if(j + 1 < N){
                    int now = check[i][j];
                    int next = check[i][j+1];
                    
                    if(now != next) {
                        list.get(now).add(next);
                        list.get(next).add(now);
                    }
                }
            }
        }
        
        int sumMax = 0;
        
        for(int i = 1; i < index; i++) {
            int prev = counts.get(i);
            int sum = list.get(i).stream().mapToInt(a -> prev + counts.get(a)).max().getAsInt();
            
            sumMax = Math.max(sum, sumMax);
        }
        
        sb.append(sumMax);
        System.out.println(sb);
    }
    
    public static int bfs(Queue<int[]> queue, int[][] check, int index) {
        int count = 0;
        int[][] dir = {{0, -1}, {-1, 0}, {0, 1}, {1, 0}};
        
        while(!queue.isEmpty()) {
            int[] now = queue.poll();
            if(check[now[0]][now[1]] > 0)
                continue;
            check[now[0]][now[1]] = index;
            count++;
            
            for(int i = 0; i < 4; i++) {
                int newA = now[0] + dir[i][0];
                int newB = now[1] + dir[i][1];
                
                if((matrix[now[0]][now[1]] >> i & 1) == 1)
                    continue;

                if(check[newA][newB] > 0)
                    continue;
                
                queue.add(new int[]{newA, newB});
            }
        }
        
        return count;
    }
}
