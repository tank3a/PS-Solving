import java.util.Arrays;

class Solution {
    public int solution(int[] picks, String[] minerals) {
        int totalAxe = 0;
        int[][] tired = {{1,1,1}, {5,1,1}, {25,5,1}};
        for(int i = 0; i < picks.length; i++)
            totalAxe += picks[i];
        
        
        if(totalAxe * 5 < minerals.length) {
            minerals = Arrays.copyOf(minerals, totalAxe * 5);
        }
        
        int[] mineralInt = new int[minerals.length];
        for(int i = 0; i < minerals.length; i++) {
            if(minerals[i].equals("diamond"))
                mineralInt[i] = 0;
            else if(minerals[i].equals("iron"))
                mineralInt[i] = 1;
            else
                mineralInt[i] = 2;
        }
        
        int totalUse = 0;
        if(mineralInt.length % 5 > 0)
            totalUse++;
        totalUse += mineralInt.length / 5;
        
        int index = 2;
        while(totalUse < totalAxe) {
            picks[index]--;
            if(picks[index] == 0)
                index--;
            
            totalAxe--;
        }

        int[] sum = new int[totalAxe];
        
        for(int i = 0; i < mineralInt.length; i++) {
            int score;
            if(mineralInt[i] == 0)
                score = 25;
            else if(mineralInt[i] == 1)
                score = 5;
            else
                score = 1;
            sum[i / 5] += score;
        }
        
        int answer = 0;

        index = 0;
        while(totalAxe > 0) {
            int temp = -1;
            int tempIdx = -1;
            for(int i = 0; i < sum.length; i++) {
                if(temp < sum[i]) {
                    tempIdx = i;
                    temp = sum[i];
                }
            }
            
            while(picks[index] <= 0) {
                index++;
            }
            
            sum[tempIdx] = 0;
            for(int i = tempIdx * 5; i < tempIdx * 5 + 5 && i < mineralInt.length; i++) {
                answer += tired[index][mineralInt[i]];
            }
            System.out.print(index + " ");
            picks[index]--;
            System.out.print(picks[index] + " ");
            System.out.println(answer);
            totalAxe--;
        }
        
        return answer;
    }
}
