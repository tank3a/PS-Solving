import java.util.*;

class Solution {
    HashSet<Integer> set = new HashSet<>();
    
    public int solution(String numbers) {
        int answer = 0;
        int len = numbers.length();
        
        int[] input = new int[len];
        for(int i = 0; i < input.length; i++) {
            input[i] = numbers.charAt(i) - '0';
        }
                
        for(int i = 0; i < len; i++) {
            boolean[] used = new boolean[len];
            used[i] = true;
            StringBuilder now = new StringBuilder();
            now.append(input[i]);
            dfs(used, input, len, now);
        }
        
        boolean[] isPrime = new boolean[(int)Math.pow(10, len)];
        isPrime[0] = true;
        isPrime[1] = true;
        for(int i = 2; i <= Math.sqrt(isPrime.length); i++) {
            if(!isPrime[i]) {
                for(int j = i * i; j < isPrime.length; j += i) {
                    isPrime[j] = true;
                }
            }
        }
        return (int)set.stream().filter(a -> !isPrime[a]).count();
    }
    
    public void dfs(boolean[] used, int[] input, int level, StringBuilder now) {
        if(level == 0)
            return;

        set.add(Integer.parseInt(now.toString()));
        for(int i = 0; i < input.length; i++) {
            if(used[i])
                continue;
            used[i] = true;
            StringBuilder next = new StringBuilder();
            next.append(now).append(input[i]);

            dfs(used, input, level - 1, next);
            used[i] = false;
        }
    }
}
