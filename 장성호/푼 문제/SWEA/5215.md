#include<iostream>
#include<algorithm>
using namespace std;

int n, l;
int dp[21][10001];
//팩토리얼은 맞는데 시간복잡도상 DP로 해야 될듯 싶은데..
int main(int argc, char** argv)
{
	int test_case;
	int T;
	cin>>T;
	for(test_case =1 ; test_case <= T; ++test_case){
        // l = 제한 칼로리
		cin >> n >> l;

		int score[21];
		int cal[21];

		for (int i = 1; i <= n; i++){
			cin >> score[i] >> cal[i];
		}
		for (int i = 1; i <= n ; i++){
	    	for (int j = 1; j <= l; j++)
			{
				if (cal[i] > j)
					dp[i][j] = dp[i - 1][j];
				else
					dp[i][j] = max(dp[i - 1][j], dp[i - 1][j - cal[i]] + score[i]);
			}
		}
		cout << "#" << test_case << " " << dp[n][l] << '\n';
	}
	return 0;//정상종료시 반드시 0을 리턴해야합니다.
}
