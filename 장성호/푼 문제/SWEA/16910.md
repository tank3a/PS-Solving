```c++
#include<iostream>

using namespace std;

int main(int argc, char** argv)
{
	int test_case;
	int T;
	cin>>T;
	
	int N;
	for(test_case = 1; test_case <= T; ++test_case)
	{
        cin>>N;
        
        
	    int answer = 0;
        for(int i=0; i<=N; i++){
            for(int j=1; j<=N; j++){
                if((i*i+j*j)<=(N*N))
                    answer++;
            }
        }
        
        cout<<"#"<<test_case<<" "<<answer*4 + 1<<endl;
	}
	return 0;//정상종료시 반드시 0을 리턴해야합니다.
}
```
