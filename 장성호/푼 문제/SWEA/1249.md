#include <iostream>
#include <queue>
using namespace std;

int T, N;
int board[101][101];
int rc_time[101][101];
bool visited[101][101];

int dx[4] = { 0,0,1,-1 };
int dy[4] = { 1,-1,0,0 };

void init()
{
	for (int i = 0; i < 101; i++) {
		for (int j = 0; j < 101; j++) {
			rc_time[i][j] = 0;
			visited[i][j] = false;
			board[i][j] = 0;
		}
	}
}

void BFS(int x, int y) {
	priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> q;
	q.push({ x,y });
	visited[0][0] = true;

	while (!q.empty()) {
		int x = q.top().first;
		int y = q.top().second;
		q.pop();

		for (int i = 0; i < 4; i++) {
			int nx = dx[i] + x;
			int ny = dy[i] + y;

			if (nx < 0 || nx >= N || ny < 0 || ny >= N) continue;
			if (!visited[nx][ny] || rc_time[nx][ny] > rc_time[x][y] + board[nx][ny]) {
				q.push({ nx,ny });
				visited[nx][ny] = true;
				rc_time[nx][ny] = rc_time[x][y] + board[nx][ny];
			}
		}
	}
}

int main() {
	cin >> T;
	for (int t = 1; t <= T; t++) {
		cin >> N;

		init();
		for (int i = 0; i < N; i++) {
			string s; cin >> s;
			for (int j = 0; j < N; j++) {
				board[i][j] = s[j] - '0';
			}
		}

		BFS(0, 0);

		cout << '#' << t << " " << rc_time[N - 1][N - 1] << '\n';
	}
}
