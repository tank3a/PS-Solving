https://www.acmicpc.net/problem/2877
```c++
#include <iostream>
#include <cmath>
using namespace std;

int k;


int main(){
    cin>>k;

    int s=1;
    int cnt=0;
    while(1){
        int tmp = pow(2, s);
        cnt += tmp;
        if(cnt>=k) {
            cnt -= tmp;
            break;
        }
        else s++;
    }
    k-=cnt;
    string ans;
    for(int i=s-1; i>=0; i--){
        int tmp = pow(2, i);
        if(tmp<k) {
            ans+= '7';
            k-=tmp;
        }
        else ans+= '4';
    }
    cout<<ans;

    return 0;
}
```
