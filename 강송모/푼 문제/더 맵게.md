import java.util.*;

class Solution {
    public int solution(int[] scoville, int K) {
        int answer = 0;
        
        PriorityQueue<Integer> sco = new PriorityQueue<>();
        for (int i = 0; i < scoville.length; i++) {
            sco.add(scoville[i]);
        }
        
        while (true) {
            Integer e1 = sco.poll();
            Integer e2 = sco.poll();
            if (e1 >= K) {
                break;
            }
            if (e1 == null || e2 == null) {
                return -1;
            }
            if (e1 == 0 && e2 == 0) {
                if (K == 0) return 0;
                else return -1;
            }
            
            int t = e1 + (e2 * 2);
            sco.add(t);
            answer += 1;
        }
        
        return answer;
    }
}
