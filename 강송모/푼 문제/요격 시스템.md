def solution(targets):
    
    targets.sort(key=lambda x:x[0])
    
    start = targets[0][0]
    end = targets[0][1]
    answer = 1
    
    for t in targets:
        if t[0] >= end:
            start = t[0]
            end = t[1]
            answer += 1
            continue
        start = t[0]
        if t[1] < end:
            end = t[1]
    
    return answer
