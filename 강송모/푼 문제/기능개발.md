import java.util.*;

class Solution {
    public int[] solution(int[] progresses, int[] speeds) {
        
        ArrayList<Integer> res = new ArrayList<>();
        
        int cr = 0;
        while (true) {
            if (cr >= (progresses.length)) break;
            
            for (int i = cr; i < progresses.length; i++) {
                progresses[i] = progresses[i] + speeds[i];
            }
            
            int amount = 0;
            for (int j = cr; j < progresses.length; j++) {
                if (progresses[j] >= 100) {
                    cr += 1;
                    amount += 1;
                } else break;
            }
            
            if (amount > 0) {
                res.add(amount);
            }
        }
        
        int[] answer = new int[res.size()];
        
        for (int k = 0; k < res.size(); k++) {
            answer[k] = res.get(k);
        }
        return answer;
    }
}
