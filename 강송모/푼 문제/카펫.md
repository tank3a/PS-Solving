class Solution {
    public int[] solution(int brown, int yellow) {
        // (x-2) * (y-2) = yellow
        // 2x + 2y - 4 = brown
        
        double x = ((brown / 2) + 2 + Math.sqrt(brown*brown/4 + 4 - 2*brown - 4*yellow)) / 2;
        double y = ((brown / 2) + 2 - Math.sqrt(brown*brown/4 + 4 - 2*brown - 4*yellow)) / 2;
        
        int[] answer = new int[2];
        
        if (x > y) {
            answer[0] = (int)x;
            answer[1] = (int)y;
        } else {
            answer[1] = (int)x;
            answer[0] = (int)y;
        }
        
        return answer;
    }
}
