import java.util.*;

class Solution {
    public int[] solution(String[] genres, int[] plays) {
        HashMap<String, int[]> map = new HashMap<>();
        
        for (int i = 0; i < plays.length; i++) {
            String tg = genres[i];
            int tp = plays[i];
            if (map.containsKey(tg)) {
                int[] temp = map.get(tg);
                // 총 수 더하기
                temp[0] = temp[0] + tp;
                // 상위 둘 변경여부
                if (temp[4] < tp) {
                    if (temp[2] < tp) {
                        temp[4] = temp[2];
                        temp[3] = temp[1];
                        temp[2] = tp;
                        temp[1] = i;
                    } else {
                        temp[4] = tp;
                        temp[3] = i;
                    }
                }
                map.put(tg, temp);
            } else {
                // 새로운 장르
                int[] temp = new int[5];
                temp[0] = tp;
                temp[1] = i;
                temp[2] = tp;
                temp[3] = -1;
                temp[4] = 0;
                map.put(tg, temp);
            }
        }
        
        Iterator<int[]> it = map.values().iterator();
        ArrayList<Integer> score = new ArrayList<>();
        HashMap<Integer, int[]> amount = new HashMap<>();
        
        int minus = 0;
        while (it.hasNext()) {
            int[] now = it.next();
            int[] t = new int[4];
            t[0] = now[1];
            t[1] = now[2];
            t[2] = now[3];
            t[3] = now[4];
            amount.put(now[0], t);
            score.add(now[0]);
            if (t[2] == -1) minus += 1;
        }
        score.sort(Comparator.reverseOrder());
        
        int[] answer = new int[score.size() * 2 - minus];
        
        int append = 0;
        for (int i = 0; i < score.size(); i++) {
            int cg = score.get(i);
            int[] res = amount.get(cg);
            answer[append] = res[0];
            append = append + 1;
            if (res[2] != -1) {
                answer[append] = res[2];
                append = append + 1;
            }
            
        }

        return answer;
    }
}
