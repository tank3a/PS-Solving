import java.util.*;

class Solution {
    
    static int cnt;
    static boolean[] visited;
    HashMap<Integer, ArrayList<Integer>> map;
    
    private void count(int loc) {
        if (!visited[loc]) {
            visited[loc] = true;
            cnt += 1;

            ArrayList<Integer> route = map.get(loc);

            for (int k = 0; k < route.size(); k++) {
                count(route.get(k));
            }
        }
    }
    
    public int solution(int n, int[][] wires) {
        
        int answer = 101;
        
        map = new HashMap<>();
        
        for (int i = 0; i < wires.length; i++) {
            if (map.containsKey(wires[i][0])) {
                ArrayList<Integer> temp = map.get(wires[i][0]);
                temp.add(wires[i][1]);
                map.put(wires[i][0], temp);
            } else {
                ArrayList<Integer> temp = new ArrayList<>();
                temp.add(wires[i][1]);
                map.put(wires[i][0], temp);
            }
            
            if (map.containsKey(wires[i][1])) {
                ArrayList<Integer> temp = map.get(wires[i][1]);
                temp.add(wires[i][0]);
                map.put(wires[i][1], temp);
            } else {
                ArrayList<Integer> temp = new ArrayList<>();
                temp.add(wires[i][0]);
                map.put(wires[i][1], temp);
            }
        }
        
        for (int j = 0; j < wires.length; j++) {
            cnt = 0;
            visited = new boolean[n+1];
            int t = wires[j][0];
            visited[wires[j][1]] = true;
            
            count(t);
            
            if (Math.abs(n - cnt - cnt) < answer) answer = Math.abs(n - cnt - cnt);
            
        }
        
        return answer;
    }
}
