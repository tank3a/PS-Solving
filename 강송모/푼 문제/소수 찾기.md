import java.util.*;

class Solution {
    
    static Set<Integer> set;
    static boolean[] visited;
    static String[] part;
    
    public int solution(String numbers) {
        set = new HashSet<>();
        visited = new boolean[numbers.length()];
        part = numbers.split("");
        
        permutation("");
        
        Iterator<Integer> it = set.iterator();
        
        int answer = 0;
        while (it.hasNext()) {
            int t = it.next();
            if (t < 2) continue;
            int cnt = 0;
            for (int i = 2; i <= Math.sqrt(t); i++) {
                if (t % i == 0) {
                    cnt += 1;
                    break;
                }
            }
            
            if (cnt == 0) answer += 1;
        }
        
        
        return answer;
    }
    
    private void permutation(String sum) {
        if (!sum.isEmpty()) {
            set.add(Integer.parseInt(sum));
        }
        
        for (int i = 0; i < part.length; i++) {
            if (!visited[i]) {
                visited[i] = true;
                permutation(sum + part[i]);
                visited[i] = false;
            }
        }
    }
}
