import java.util.*;

class Solution {
    public int solution(int[] priorities, int location) {
        int answer = 1;
        
        ArrayList<Integer> index = new ArrayList<>();
        ArrayList<Integer> prior = new ArrayList<>();
        
        for (int i = 0; i < priorities.length; i++) {
            int a = priorities[i];
            index.add(a);
            prior.add(a);
        }
        prior.sort(Comparator.reverseOrder());
        
        while (true) {
            int ind = index.get(0);
            int pri = prior.get(0);

            if (location == 0) {
                if (ind == pri) {
                    break;
                } else if (ind < pri) {
                    index.remove(0);
                    index.add(ind);
                    location = index.size() - 1;
                }
            } else {
                location -= 1;
                if (ind == pri) {
                    index.remove(0);
                    prior.remove(0);
                    answer += 1;
                } else if (ind < pri) {
                    index.remove(0);
                    index.add(ind);
                }
            }
            
            
        }
        
        return answer;
    }
}
