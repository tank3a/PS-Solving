import java.util.*;

class Solution {
    public boolean solution(String[] phone_book) {

        HashMap<String, Boolean> book = new HashMap<>();
        
        for (int i = 0; i < phone_book.length; i++) {
            
            book.put(phone_book[i], true);
        }
        
        for (int k = 0; k < phone_book.length; k++) {
            String te = phone_book[k];
            
            for (int h = 1; h < te.length(); h++) {
                
                if (book.containsKey(te.substring(0, h))) {
                    return false;
                }
            }
        }
        
        return true;
    }
}
