import java.util.*;

class Solution {
    
    static boolean[] visited;
    static int[][] dungeon;
    
    private int go(int cnt, int rest) {
        int max = cnt; // 방문횟수
        
        for (int i = 0; i < dungeon.length; i++) {
            int cost = 0;
            
            if (!visited[i]) {
                visited[i] = true;
                if (rest >= dungeon[i][0]) {
                    cost = go(cnt + 1, rest - dungeon[i][1]);
                } else {
                    visited[i] = false;
                    continue;
                }
                visited[i] = false;
            }
            
            if (cost > max) max = cost;
        }
        
        return max;
    }
    
    public int solution(int k, int[][] dungeons) {
        visited = new boolean[dungeons.length];
        dungeon = dungeons;
        
        int answer = go(0, k);
        
        return answer;
    }
}
