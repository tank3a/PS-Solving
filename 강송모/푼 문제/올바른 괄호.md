class Solution {
    boolean solution(String s) {
        boolean answer = true;

        int ac = 0;
        int cnt = 0;
        
        for (int i = 0; i < s.length(); i++) {
            char t = s.charAt(i);
            if (t == '(') {
                cnt += 1;
            } else {
                if (cnt == 0) {
                    answer = false;
                    break;
                }
                cnt -= 1;
            }
        }
        
        if (cnt != 0) answer = false;

        return answer;
    }
}
