import java.util.*;

class Solution {
    public int[] solution(String[] operations) {
        int[] answer = new int[2];
        
        PriorityQueue<Integer> maxq = new PriorityQueue<>(Comparator.reverseOrder());
        PriorityQueue<Integer> minq = new PriorityQueue<>();
        
        int amount = 0;
        for (int i = 0; i < operations.length; i++) {
            String t = operations[i];
            char ins = t.charAt(0);
            
            if (ins == 'I') {
                int ad = Integer.parseInt(t.substring(2));
                maxq.add(ad);
                minq.add(ad);
                amount += 1;
            }
            if (ins == 'D') {
                if (amount <= 0) {
                    continue;
                }
                char op = t.charAt(2);
                if (op == '-') {
                    Integer temp = minq.poll();
                    maxq.remove(temp);
                } else {
                    Integer temp = maxq.poll();
                    minq.remove(temp);
                }
                amount -= 1;
            }
        }
        
        if (amount == 0) {
            answer[0] = 0;
            answer[1] = 0;
        } else {
            answer[0] = maxq.poll();
            answer[1] = minq.poll();
        }
        
        return answer;
    }
}ㄴ
