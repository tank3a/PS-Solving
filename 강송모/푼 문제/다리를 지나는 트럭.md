import java.util.*;

class Solution {
    public int solution(int bridge_length, int weight, int[] truck_weights) {
        
        ArrayList<Integer> bridge = new ArrayList<>();
        for (int i = 0; i < bridge_length; i++) {
            bridge.add(0);
        }
        
        int answer = 0;
        int tindex = 0;
        int iweight = weight;
        
        while (true) {
            if (tindex >= truck_weights.length && iweight == weight) {
                break;
            }
            int t = 10001;
            if (tindex < truck_weights.length) {
                t = truck_weights[tindex];
            }
            
            int arr = bridge.get(0);
            if (arr > 0) {
                weight = weight + arr;
            }
            bridge.remove(0);
            answer += 1;
            
            if (t <= weight) {
                bridge.add(t);
                weight = weight - t;
                tindex += 1;
            } else {
                bridge.add(0);
            }
        }
        
        return answer;
    }
}
