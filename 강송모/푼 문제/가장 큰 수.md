import java.util.*;

class Solution {
    public String solution(int[] numbers) {
        
        Integer[] res = new Integer[numbers.length];
        
        for (int i = 0; i < numbers.length; i++) {
            res[i] = numbers[i];
        }
        
        Arrays.sort(res, new Comparator<Integer>() {
            public int compare(Integer i1, Integer i2) {
                String pres = Integer.toString(i1);
                String posts = Integer.toString(i2);
                
                int a = Integer.parseInt(pres+posts);
                int b = Integer.parseInt(posts+pres);
                
                if (a > b) {
                    return -1;
                } else if(b > a) return 1;
                else return 0;
            }
        });
        
        StringBuilder sb = new StringBuilder();
        
        for (int k = 0; k < res.length; k++) {
            sb.append(res[k]);
        }
        
        String answer = sb.toString();
        
        if (answer.charAt(0) == '0') {
            answer = "0";
        }
        
        return answer;
    }
}
