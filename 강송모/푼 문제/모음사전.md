class Solution {
    public int solution(String word) {
        int answer = 1;
        
        if (word.charAt(0) == 'E') {
            answer += 781;
        } else if (word.charAt(0) == 'I') {
            answer += 1562;
        } else if (word.charAt(0) == 'O') {
            answer += 2343;
        } else if (word.charAt(0) == 'U') {
            answer += 3124;
        }
        
        if (word.length() > 1) {
            answer += 1;
            if (word.charAt(1) == 'E') {
                answer += 156;
            } else if (word.charAt(1) == 'I') {
                answer += 312;
            } else if (word.charAt(1) == 'O') {
                answer += 468;
            } else if (word.charAt(1) == 'U') {
                answer += 624;
            }
        }
        
        if (word.length() > 2) {
            answer += 1;
            if (word.charAt(2) == 'E') {
                answer += 31;
            } else if (word.charAt(2) == 'I') {
                answer += 62;
            } else if (word.charAt(2) == 'O') {
                answer += 93;
            } else if (word.charAt(2) == 'U') {
                answer += 124;
            }
        }
        
        if (word.length() > 3) {
            answer += 1;
            if (word.charAt(3) == 'E') {
                answer += 6;
            } else if (word.charAt(3) == 'I') {
                answer += 12;
            } else if (word.charAt(3) == 'O') {
                answer += 18;
            } else if (word.charAt(3) == 'U') {
                answer += 24;
            }
        }
        
        if (word.length() > 4) {
            answer += 1;
            if (word.charAt(4) == 'E') {
                answer += 1;
            } else if (word.charAt(4) == 'I') {
                answer += 2;
            } else if (word.charAt(4) == 'O') {
                answer += 3;
            } else if (word.charAt(4) == 'U') {
                answer += 4;
            }
        }
        
        return answer;
    }
}
