#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

bool range[20 * 100000];
int S[20];
int N;
void dfs(int begin, int sum)
{
if (begin == N) {
range[sum] = true;
return;
}
dfs(begin + 1, sum + S[begin]);
dfs(begin + 1, sum);
}
int main(){
cin >> N;
for (int i = 0; i < N; ++i){
cin >> S[i];
}

    dfs(0, 0);

    int start = 0;
    while (true){
    	if (range[start] != 1) {
    		cout << start << "\n";
    		return 0;
    	}
    	start++;
    }

}
