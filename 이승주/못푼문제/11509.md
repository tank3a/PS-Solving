#include <iostream>
using namespace std;

int N,x,MAX;
int arr[1000005];
bool isVisited[1000005];

int main(){

    cin>>N;

    for(int i=0; i<N; i++) {
        cin>>x;

        if(arr[x+1]>0) arr[x+1]--;
        arr[x]++;
        if(MAX<x) MAX=x;
    }

    int ans=0;
    for(int i=1; i<=MAX; i++) ans+=arr[i];

    cout<<ans;
