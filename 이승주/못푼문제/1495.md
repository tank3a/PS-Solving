#include <iostream>
#include <algorithm>

using namespace std;
int n, s, m;
int arr[51];
bool dp[51][1001];

int main() {
ios_base::sync_with_stdio(false);
cin.tie(NULL), cout.tie(NULL);
cin >> n >> s >> m;
for(int i=1;i<=n;i++)
cin >> arr[i];
if(s+arr[1] <= m)
dp[1]s+arr[1]] = true;
if(s-arr[1] >= 0)
dp[1]s-arr[1]] = true;
for(int i=2;i<=n;i++){
for(int j=0;j<=m;j++){
if(dp[i-1][j] == true){
if(j + arr[i] <= m)
dp[i]j+arr[i]] = true;
if(j - arr[i] >= 0)
dp[i]j-arr[i]] = true;
}
}
}
for(int i=m; i>=0; i--){
if(dp[n][i] == true){
cout << i;
return 0;
}
}
cout << -1;
}
