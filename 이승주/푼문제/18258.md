#include <iostream>
#include <queue>
#include <string>

using namespace std;

int main() {
  ios_base::sync_with_stdio(false);
  cin.tie(NULL);
  cout.tie(NULL);
  int N;
  cin >> N;
  queue<int> q;
  for(int i=0;i<N;i++){
    string str;
    cin >> str;
    if(str == "push"){
      int num;
      cin >> num;
      q.push(num);
    }
    else if(str == "pop"){
      if(q.size() == 0)
        cout << -1 << '\n';
      else{
        cout << q.front() << '\n';
        q.pop();
      }
    }
    else if(str == "size")
      cout << q.size() << '\n';
    else if(str == "empty"){
      if(q.size()==0)
        cout << 1 << '\n';
      else
        cout << 0 << '\n';
    }
    else if(str == "front"){
      if(q.size() == 0)
        cout << -1 << '\n';
      else
        cout << q.front() << '\n';
    }
    else if(str == "back"){
      if(q.size() == 0)
        cout << -1 << '\n';
      else
        cout << q.back() << '\n';
    }
  }
}
