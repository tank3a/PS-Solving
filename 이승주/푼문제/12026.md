#include <iostream>
#include <algorithm>

using namespace std;
int n;
char road[1001];
int dp[1001];

int main() {
ios_base::sync_with_stdio(false);
cin.tie(NULL), cout.tie(NULL);
cin >> n;
for(int i=1;i<=n;i++)
cin >> road[i];
fill(dp, dp+1001, 9999999);
dp[0] = 0, dp[1] = 0;
for(int i=1;i<n;i++){
if(road[i] == 'B'){
for(int j=i+1;j<=n;j++){
if(road[j] == 'O')
dp[j] = min(dp[j], (j-i)_(j-i) + dp[i]);
}
}
else if(road[i] == 'O'){
for(int j=i+1;j<=n;j++){
if(road[j] == 'J')
dp[j] = min(dp[j], (j-i)_(j-i) + dp[i]);
}
}
else{
for(int j=i+1;j<=n;j++){
if(road[j] == 'B')
dp[j] = min(dp[j], (j-i)\*(j-i) + dp[i]);
}
}
}
if(dp[n] == 9999999)
cout << -1;
else
cout << dp[n];
}
